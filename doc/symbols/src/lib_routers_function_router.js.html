 18 var url = require('url')
 19   , BaseRouter = require('./base_router').BaseRouter;
 20 
 21 var FunctionRouter = function (routes) {
 22   this.routes = [];
 23   if (routes) {
 24     this.setRoutes(routes);
 25   }
 26 };
 27 
 28 FunctionRouter.prototype = new BaseRouter();
 29 
 30 FunctionRouter.prototype.first = function (req) {
 31   var routes = this.routes
 32     , route
 33     , handler
 34     , params = url.parse(req.url, true).query;
 35   for (var i = 0, ii = routes.length; i < ii; i++) {
 36     route = routes[i];
 37     if (route.pattern(req, params)) {
 38       handler = route.handler;
 39  40       params.controller = handler.controller;
 41       params.action = handler.action;
 42       return params;
 43     }
 44   }
 45   return null;
 46 };
 47 
 48 exports.FunctionRouter = FunctionRouter;
