 18 var string = require('./string').string
 19   , async = require('./async').async
 20   , uri = require('./uri').uri;
 21 
 22 var utils = new (function () {
 23 
 24   var _mix = function (targ, src, merge, includeProto) {
 25     for (var p in src) {
 26       // Don't copy stuff from the prototype
 27  28       if (src.hasOwnProperty(p) || includeProto) {
 29         if (merge &&
 30             // Assumes the source property is an Object you can
 31             // actually recurse down into
 32             (typeof src[p] == 'object') &&
 33             (src[p] !== null) &&
 34             !(src[p] instanceof Array)) {
 35           // Create the source property if it doesn't exist
 36           // TODO: What if it's something weird like a String or Number?
 37           if (typeof targ[p] == 'undefined') {
 38             targ[p] = {};
 39           }
 40           _mix(targ[p], src[p], merge, includeProto); // Recurse
 41         }
 42         // If it's not a merge-copy, just set and forget
 43         else {
 44           targ[p] = src[p];
 45         }
 46       }
 47     }
 48   };
 49 
 50   this.objectToString = function (object) {
 51     var objectArray = [];
 52     for (var key in object) {
 53       if ('object' == typeof object[key]) {
 54         objectArray.push(this.objectToString(object[key]));
 55       } else {
 56         objectArray.push(key + '=' + object[key]);
 57       }
 58     }
 59     return objectArray.join(', ');
 60   };
 61 
 62   /*
 63    * Mix in the properties on an object to another object
 64    * yam.mixin(target, source, [source,] [source, etc.] [merge-flag]);
 65    * 'merge' recurses, to merge object sub-properties together instead
 66    * of just overwriting with the source object.
 67    */
 68   this.mixin = (function () {
 69     return function () {
 70       var args = Array.prototype.slice.apply(arguments),
 71           merge = false,
 72           targ, sources;
 73       if (args.length > 2) {
 74         if (typeof args[args.length - 1] == 'boolean') {
 75           merge = args.pop();
 76         }
 77       }
 78       targ = args.shift();
 79       sources = args;
 80       for (var i = 0, ii = sources.length; i < ii; i++) {
 81         _mix(targ, sources[i], merge);
 82       }
 83       return targ;
 84     };
 85   }).call(this);
 86 
 87   this.enhance = (function () {
 88     return function () {
 89       var args = Array.prototype.slice.apply(arguments),
 90           merge = false,
 91           targ, sources;
 92       if (args.length > 2) {
 93         if (typeof args[args.length - 1] == 'boolean') {
 94           merge = args.pop();
 95         }
 96       }
 97       targ = args.shift();
 98       sources = args;
 99       for (var i = 0, ii = sources.length; i < ii; i++) {
100         _mix(targ, sources[i], merge, true);
101       }
102       return targ;
103     };
104   }).call(this);
105 
106 })();
107 
108 utils.MessageParser = function (dispatch, dispatchScope, notParsedHandler) {
109   this._data = '';
110   this.dispatch = dispatch;
111   this.dispatchScope = dispatchScope;
112   this.notParsedHandler = notParsedHandler;
113 };
114 
115 utils.MessageParser.prototype = new (function () {
116   this.handle = function (d) {
117     var arr
118       , msg;
119 
120     this._data += d;
121     arr = this._data.split('\n');
122     for (var i = 0, ii = (arr.length - 1); i < ii; i++) {
123       if (arr[i]) {
124         msg = null;
125         try {
126           msg = JSON.parse(arr[i]);
127         }
128         catch (e) {}
129 
130         if (msg && msg.method && this.dispatch[msg.method]) {
131           this.dispatch[msg.method].call(this.dispatchScope, msg);
132         }
133         else {
134           if (typeof this.notParsedHandler == 'function') {
135             this.notParsedHandler(arr[i]);
136           }
137         }
138       }
139     }
140     this._data = (arr[arr.length - 1] || '');
141   };
142 })();
