 19 // need this, but maybe there's a better spot for it?
 20 var regExpEscape = (function() {
 21   var specials = [ '/', '.', '*', '+', '?', '|', '(', ')', '[', ']', '{', '}', '\\' ];
 22   sRE = new RegExp('(\\' + specials.join('|\\') + ')', 'g');
 23   return function (text) { return text.replace(sRE, '\\$1'); };
 24 })();

 26 /*
 27  * Simple router for Node -- setting up routes looks like this:
 28  *
 29  * r = new Router();
 30  * r.match('/').to({ controller: 'Main', action: 'index' }).name('main');
 31  * r.match('/users/:userid/messages/:messageid').to({controller: 'Users', action: 'index'});
 32  *
 33  * Pretty familiar to anyone who's used Merb/Rails
 34  */
 35 
 36 /*
 37  * Router - it routes.
 38  */
 39 var Router = function () {
 40   this.routes = [];
 41   
 42   /*
 43    * Router.match()
 44    *
 45    * r.match('/:controller/:action(/:id)(.:format)', 'GET',{ id:/\d+/ }).to(......)
 46    *
 47    * path is mandatory (duh)
 48    * method is optional, routes without a method will apply in all cases
 49    * the options hash (which is also optional) compares either a regex or a
 50    * string to the key's value in the path
 51    *
 52    * returns: Route (for chaining)
 53    */
 54   this.match = function () {
 55     var path, method, opts, args, arg;
 56     args = Array.prototype.slice.call(arguments);
 57     for (var i = 0, ii = args.length; i < ii; i++) {
 58       arg = args[i];
 59       if (i === 0) { // path is always first
 60         path = arg;
 61       } else if (typeof(arg) == 'string') { // method is the only other string
 62         method = arg;
 63       } else { // opts hash
 64         opts = arg || {};
 65       }
 66     }
 67     
 68     route = new Route(path, method, opts);
 69     this.routes.push(route);
 70 
 71     // TODO: set route options & regex
 72     
 73     return route;
 74   };
 75 
 76   /*
 77    * Router.resource()
 78    *
 79    * r.resource('Products')
 80    *
 81    * generates standard resource routes for a controller name
 82    *
 83    * returns: array of Routes (for now)
 84    */
 85   this.resource = function (res) {
 86     var controller = geddy.string.camelize(res, true);
 87     var cslug = geddy.string.decamelize(res);
 88     return [
 89       this.match('/'+cslug+'(.:format)', 'GET')
 90           .to({ controller: controller, action: 'index' }),
 91       this.match('/'+cslug+'/add(.:format)', 'GET')
 92           .to({ controller: controller, action: 'add' }),
 93       this.match('/'+cslug+'/:id(.:format)', 'GET')
 94           .to({ controller: controller, action: 'show' }),
 95       this.match('/'+cslug+'/:id/edit(.:format)', 'GET')
 96           .to({ controller: controller, action: 'edit' }),
 97       this.match('/'+cslug+'(.:format)', 'POST')
 98           .to({ controller: controller, action: 'create' }),
 99       this.match('/'+cslug+'/:id(.:format)', 'PUT')
100           .to({ controller: controller, action: 'update' }),
101       this.match('/'+cslug+'/:id(.:format)', 'DELETE')
102           .to({ controller: controller, action: 'remove' })
103     ];
104   };
105   
106   // find the first route that match the path & method
107   /*
108    * geddy.router.first('/snow_dogs/5', 'GET')
109    * => { controller: 'SnowDogs', action: 'show', id: 5, method: 'GET' }
110    *
111    * find & return a params hash from the first route that matches
112    *
113    * if there's no match, this will return false
114    */
115   this.first = function (req) {
116     var route = false
117       , path = req.url
118       , method = req.method;
119     for (var i in this.routes) {
120       // if the method doesn't match: jog on
121       if (typeof(this.routes[i].method) != 'undefined' && this.routes[i].method != method) {
122           continue;
123       }
124       // attempt the parse
125       route = this.routes[i].parse(path, method);
126       if (route) {
127         return route;
128       }
129     }
130     return false;
131   };
132   
133   /*
134    * geddy.router.all('/snow_dogs/bob', 'GET') => 
135    *  [
136    *    { controller: 'SnowDogs', action: 'show', id: 'bob', method: 'GET' },
137    *    { controller: 'SnowDogs', action: 'bob', method: 'GET' }
138    *  ]
139    *
140    * find & return a params hash from ALL routes that match
141    *
142    * if there's no matches, returns an empty array
143    */
144   this.all = function (path, method) {
145     var ret = [];
146     var route = false;
147     for (var i in this.routes) {
148       // if the method doesn't match: jog on
149       if (typeof(this.routes[i].method) != 'undefined' && this.routes[i].method != method) {
150         continue;
151       }
152       // attempt the parse
153       route = this.routes[i].parse(path, method);
154       if (route) {
155         ret.push(route);
156       }
157     }
158     return ret;
159   };
160   
161   /*
162    * geddy.router.url({ controller: 'SnowDogs', action: 'show', id: 5 }) => '/snow_dogs/5'
163    *  
164    * geddy.router.url({ controller: 'SnowDogs', action: 'show', id: 5, format: 'json' })
165    *    => '/snow_dogs/5.json'
166    *  
167    * generates a URL from a params hash
168    *
169    * returns false if there are no suitable routes
170    */
171   this.url = function (params) {
172     var url = false;
173         
174     // iterate through the existing routes until a suitable match is found
175     for (var i in this.routes) {
176       // do the controller & acton match?
177       if (typeof(this.routes[i].params.controller) != 'undefined' &&
178           this.routes[i].params.controller != params.controller) {
179         continue;
180       }
181       if (typeof(this.routes[i].params.action) != 'undefined' &&
182           this.routes[i].params.action != params.action) {
183         continue;
184       }
185       // attempt the stringification with defaults mixed in
186       params = geddy.mixin({controller:'Application', action:'index' }, params);
187       url = this.routes[i].stringify(params);
188       if (url) {
189         break;
190       }
191     } 
192 
193     // no love? return false
194     if (!url) {
195       return false;
196     }
197     
198     /*
199     // build the possibly empty query string
200     var qs = require('querystring').stringify(url[1]);
201     
202     // if there is a query string...
203     if (qs.length > 0) {
204       return url[0] + '?' + qs;
205     }
206     */
207     
208     // just return the url
209     return url[0];
210   };
211 
212   /*
213    * Route - turns strings into magical ponies that come when you call them
214    *
215    * ex: route = Route('/:controller/:action/:id(.:format)')
216    * ex: route = Route('/:controller/:action(/:id)(.:format)', 'GET')
217    * ex: route = Route('/:controller/:action(/:id)(.:format)',
218    *   { controller:'snow_dogs', acton:'show' })
219    * ex: route = Route('/:controller/:action/:id(.:format)', 'GET', { id:/\d+/ })
220    *
221    * Pretty familiar to anyone who's used Merb/Rails - called by Router.match()
222    */
223   var Route = function () {
224     
225     // regexen crossing
226     // matches keys, BEWARE: keys containing a "-" will need to be references
227     // in params['key-name'] notation
228     var KEY = /:([a-zA-Z_\-][\w\-]*)/;
229     // optional group (the part in parens)
230     var OGRP = /\(([^)]+)\)/;
231     // breaks a string into atomic parts: ogrps, keys, then everything else
232     var PARTS = /\([^)]+\)|:[a-zA-Z_][\w\-]*|[\w\-_\\\/\.]+/g;
233     var METHODS = /GET|POST|PUT|DELETE/;
234 
235     // parse th'args
236     var path;
237     var opts = {};
238     this.optional = false; // for nested optional url segments like (.:format)
239     
240     var args = Array.prototype.slice.call(arguments);
241     var arg;
242     for (var i = 0, ii = args.length; i < ii; i++) {
243       arg = args[i];
244       if (i === 0) { // path is always first
245         path = arg;
246       } else if (typeof(arg) == 'string') { // method is the only other string
247         this.method = arg.toUpperCase();
248       } else if (typeof(arg) == 'boolean') { // optional
249         this.optional = arg;
250       } else { // opts hash
251         opts = arg || {};
252       }
253     }
254     
255     // route-specific params
256     this.params = {};
257     this.parts = [];
258     this.name = null;
259 
260     // returns a composite regex string of all route parts
261     this.regexString = function() {
262       var ret = '';
263       // build master regex from params hash
264       for (var i in this.parts) {
265         var part = this.parts[i];
266         if (part instanceof Key) {
267           ret += part.regexString();
268         } else if (part instanceof Route) {
269           ret += part.regexString();
270         } else { // string
271           ret += regExpEscape(part);
272         }
273       }
274       if (this.optional) {
275         return '('+ret+')?';
276       }
277       return '('+ret+')';
278     };
279   
280     // builds & tests on a full regex of the entire path
281     this.test = function (string) {
282       return RegExp('^' + this.regexString() + '(\\\?.*)?$').test(string);
283     };
284 
285     /*
286      * Route.to({ controller: 'SnowDogs', action: 'index' })
287      *
288      * just mixes in route params
289 290      *
291      * returns the Route for chaining
292      */
293     this.to = function (params) {
294       geddy.mixin(this.params, params);
295       return this; // chainable
296     };
297 
298     /*
299      * Route.name('ponies')
300      *
301      * just sets the route name - NAMED ROUTES ARE NOT CURRENTLY USED
302      *
303      * returns: the Route for chaining
304      */
305     this.name = function (name) {
306       this.name = name;
307       return this; // chainable
308     };
309     
310     /*
311      * Route.stringify(params)
312      *
313      * builds a string url for this Route from a params object
314      *
315      * returns: [ "url", [leftover params] ]
316      *
317      * ** this is meant to be called & modified by Router.url()
318      */
319     this.stringify = function (params) {
320       var url = []; // urls start life as an array to enble a second pass
321       
322       for (var i in this.parts) {
323         var part = this.parts[i];
324         if (part instanceof Key) {
325           if (typeof(params[part.name]) != 'undefined' &&
326               part.regex.test(params[part.name])) {
327             // there's a param named this && the param matches the key's regex
328             url.push(part.url(params[part.name])); // push it onto the stack 
329             delete params[part.name]; // and remove from list of params              
330           } else if (this.optional) {
331             // (sub)route doesn't match, move on
332             return false;
333           }
334         } else if (part instanceof Route) {
335           // sub-routes must be handled in the next pass
336           // to avoid leftover param duplication
337           url.push(part);
338         } else { // string
339           url.push(part);
340         }
341       }
342       
343       // second pass, resolve optional parts
344       for (var i in url) {
345         if (url[i] instanceof Route) { 
346           url[i] = url[i].stringify(params); // recursion is your friend
347           // it resolved to a url fragment!
348           if (url[i]) {
349             // replace leftover params hash with the new, smaller leftover params hash
350             params = url[i][1];
351             // leave only the string for joining            
352             url[i] = url[i][0];
353           } else {
354             delete url[i]; // get rid of these shits
355           }
356         }
357       }
358       
359       for (var i in this.params) {
360         // remove from leftovers, they're implied in the to() portion of the route
361         delete params[i];
362       }
363  
364       return [url.join(''), params];
365     }; 
366     
367     this.keysAndRoutes = function () {
368       // Keys and routes, not Kernighan and Ritchie
369       var knr = [];
370 371       for (var i in this.parts) {
372         if (this.parts[i] instanceof Key || this.parts[i] instanceof Route) {
373           knr.push(this.parts[i]);
374         }
375       }
376       return knr;
377     };
378 
379     this.keys = function () {
380       var keys = [];
381       for (var i in this.parts) {
382         if (this.parts[i] instanceof Key) {
383           keys.push(this.parts[i]);
384         }
385       }
386       return keys;
387     };
388 
389     this.keyNames = function () {
390       var keys = this.keys();
391       var names = [];
392       for (var i in keys) {
393         names.push(keys[i].name);
394       }
395       return names;
396     };
397 
398     /*
399      * parses a URL into a params object
400      *
401      * route.parse(url, method)
402      *
403      * returns: a params hash || false (if the route doesn't match)
404      *
405      * ** this is meant to be called by Router.first() && Router.all() 
406      */
407     this.parse = function (urlParam, method) {
408       
409       // parse the URL with the regex & step along with the parts, 
410       // assigning the vals from the url to the names of the keys as we go (potentially stoopid)
411       
412       // let's chop off the QS to make life easier
413       var path = urlParam.split(/\?|;/)[0];
414       var params = {method:method};
415       
416       for (var key in this.params) { params[key] = this.params[key]; } 
417 
418       // if the method doesn't match, gtfo immediately
419       if (typeof(this.method) != 'undefined' && this.method != params.method) {
420         return false;
421       }
422 
423       // if the route doesn't match the regex, gtfo
424       if (!this.test(path)) {
425         return false;
426       }
427 
428       // parse the URL with the regex
429       var parts = new RegExp('^' + this.regexString() + '$').exec(path);
430       var j = 2; // index of the parts array, starts at 2 to bypass the entire match string & the entire match
431 
432       var keysAndRoutes = this.keysAndRoutes();      
433 
434       for (var i in keysAndRoutes) {        
435         if (keysAndRoutes[i] instanceof Key) {
436           if (keysAndRoutes[i].test(parts[j])) {
437             params[keysAndRoutes[i].name] = parts[j];
438           }
439         } else if (keysAndRoutes[i] instanceof Route) {
440           if (keysAndRoutes[i].test(parts[j])) {
441             // parse the subroute
442             var subparams = keysAndRoutes[i].parse(parts[j], method);
443             geddy.mixin(params, subparams);
444             // advance the parts pointer by the number of submatches
445             j+= parts[j].match(keysAndRoutes[i].regexString()).length-2 || 0;
446           } else {
447             j++; 
448           }
449         }
450         j++;
451       }
452       
453       // camelize the controller
454       if (params.controller) {
455         params.controller = geddy.string.camelize(params.controller, true);
456       }
457 
458       return params;
459     };
460 
461 462     /*
463      * keys - the building blocks of routes
464      *
465      * ex: key = new Key('name')
466      * ex: key = new Key('name', true)
467      * ex: key = new Key('name', /\w+/, rue)
468      * ex: key = new Key('name', 'ted')
469      * ex: key = new Key('id', /\d+/)
470      *
471      */
472     var Key = function () {
473 
474       // parse th'args
475       this.optional = false;
476       this.regex = /[\w\-]+/; // default url-friendly regex
477 
478       var args = Array.prototype.slice.call(arguments);
479 480       var arg;
481       for (var i = 0, ii = args.length; i < ii; i++) {
482         arg = args[i];
483         if (i === 0) { // name is always first
484           this.name = arg;
485         } 
486         // optional flag - *for future use* : /route/? would make the trailing slash optional
487         else if (typeof(arg) == 'boolean') {
488 489           this.optional = arg;
490         }
491         // regex match
492         else if (arg instanceof RegExp) {
493           this.regex = arg;
494         } 
495         // must be a regex in string-ish form
496         else {
497           this.regex = RegExp(arg);
498         }
499       }
500 
501       // special defaults for controllers & actions, which will always be function names
502       if (this.name == 'controller' || this.name == 'action') {
503         this.regex = /[a-zA-Z_][\w\-]+/;
504       }
505 
506       // makes a regex string of the key - used by this.test()
507       this.regexString = function () {
508         var ret = String(this.regex).replace(/^\//, '').replace(/\/[gis]?$/, '');
509         if (this.optional) {
510           return '(' + ret + ')?';
511         }
512         return '(' + ret + ')';
513       };
514 
515       // validates a string using the key's regex pattern
516       this.test = function (string) {
517         return new RegExp('^'+this.regexString()+'$').test(string);
518       };
519 
520       this.url = function (string) {
521         if (this.test(string)) {
522           // snake_caseify the controller, if there is one
523           if (this.name == 'controller') {
524             return geddy.string.decamelize(string);
525           }
526           return string;
527         }
528         return false; // doesn't match, let it go
529       };
530 
531       return this; // don't know if this is needed, since 'new Key' implies it 
532     };
533 
534     // path parsing
535     while (part = PARTS.exec(path)) {
536       this.parts.push(part);
537     }
538     
539     // have to do this in two passes due to RegExp execution limits    
540     for (var i in this.parts) {
541       if (OGRP.test(this.parts[i])) { // optional group
542         this.parts[i] = new Route(OGRP.exec(this.parts[i])[1], true, opts);
543       } else if(KEY.test(this.parts[i])) { // key
544         var keyname = KEY.exec(this.parts[i])[1];
545         if(opts[keyname]) { // are there match conditions for this key?
546           this.parts[i] = new Key(keyname, opts[keyname]);
547         } else{
548           this.parts[i] = new Key(keyname);
549         }        
550       } else{ // string
551         this.parts[i] = String(this.parts[i]);
552       }
553     }  
554 
555     return this;
556 
557   }; // Route
558 
559 }; // Router
560 
561 module.exports.RegExpRouter = Router;
