 18 var string = new (function () {
 19   // Regexes used in trimming functions
 20   var _LTR = /^\s+/;
 21   var _RTR = /\s+$/;
 22   var _TR = /^\s+|\s+$/g;
 23   var _NL = /\n|\r|\r\n/g;
 24   // From/to char mappings -- for the XML escape,
 25   // unescape, and test for escapable chars
 26   var _CHARS = {
 27     '&': '&',
 28     '<': '<',
 29     '>': '>',
 30     '"': '"',
 31     '\'': '''
 32   };
 33   var _UUID_CHARS = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz'.split('');
 34   // Builds the escape/unescape methods using a common
 35   // map of characters
 36   var _buildEscapes = function (direction) {
 37     return function (str) {
 38       s = str;
 39       var fr, to;
 40       for (var p in _CHARS) {
 41         fr = direction == 'to' ? p : _CHARS[p];
 42         to = direction == 'to' ? _CHARS[p] : p;
 43         s = s.replace(new RegExp(fr, 'gm'), to);
 44       }
 45       return s;
 46     };
 47   };
 48   // Builds a method that tests for any of the escapable
 49   // characters -- useful for avoiding double-escaping if
 50   // you're not sure whether a string is already escaped
 51   var _buildEscapeTest = function (direction) {
 52     return function (s) {
 53       var pat = '';
 54       for (var p in _CHARS) {
 55         pat += direction == 'to' ? p : _CHARS[p];
 56         pat += '|';
 57       }
 58       pat = pat.substr(0, pat.length - 1);
 59       pat = new RegExp(pat, "gm");
 60       return pat.test(s);
 61     };
 62   };
 63 
 64   // Escape special chars to entities
 65   this.escapeXML = _buildEscapes('to');
 66 
 67   // Unescape entities to special chars
 68   this.unescapeXML = _buildEscapes('from');
 69 
 70   // Test if a string includes special chars that
 71   // require escaping
 72   this.needsEscape = _buildEscapeTest('to');
 73 
 74   this.needsUnescape = _buildEscapeTest('from');
 75 
 76   this.toArray = function (str) {
 77     var arr = [];
 78     for (var i = 0; i < str.length; i++) {
 79       arr[i] = str.substr(i, 1);
 80     }
 81     return arr;
 82   };
 83 
 84   this.reverse = function (str) {
 85     return this.toArray(str).reverse().join('');
 86   };
 87 
 88   this.ltrim = function (str, chr) {
 89     var pat = chr ? new RegExp('^' + chr + '+') : _LTR;
 90     return str.replace(pat, '');
 91   };
 92 
 93   this.rtrim = function (str, chr) {
 94     var pat = chr ? new RegExp(chr + '+$') : _RTR;
 95     return str.replace(pat, '');
 96   };
 97 
 98   this.trim = function (str, chr) {
 99     var pat = chr ? new RegExp('^' + chr + '+|' + chr + '+$', 'g') : _TR;
100     return str.replace(pat, '');
101   };
102 
103   this.lpad = function (str, chr, width) {
104     var s = str;
105     while (s.length < width) {
106       s = chr + s;
107     }
108     return s;
109   };
110 
111   this.rpad = function (str, chr, width) {
112     var s = str;
113     while (s.length < width) {
114       s = s + chr;
115     }
116     return s;
117   };
118 
119   this.nl2br = function (str) {
120 	  return str.replace(_NL,'<br />');
121   };
122 
123   // Converts someVariableName to some_variable_name
124   this.snakeize = function (s) {
125     return s.replace(/([A-Z]+)/g, '_$1').toLowerCase().
126       replace(/^_/, '');
127   };
128 
129   // Backward-compat
130   this.decamelize = this.snakeize;
131 
132   // Converts some_variable_name to someVariableName or SomeVariableName
133   this.camelize = function (s, initialCap) {
134     var ret = s.replace(/_[a-z]{1}/g, function (s)
135       { return s.replace('_', '').toUpperCase() });
136     if (initialCap) {
137       ret = this.capitalize(ret);
138     }
139     return ret;
140   };
141 
142   this.capitalize = function (s) {
143     return s.substr(0, 1).toUpperCase() + s.substr(1);
144   };
145 
146   // From Math.uuid.js, http://www.broofa.com/Tools/Math.uuid.js
147   // Robert Kieffer (robert@broofa.com), MIT license
148   this.uuid = function () {
149     var chars = _UUID_CHARS, uuid = [], rnd=0, r;
150     for (var i = 0; i < 36; i++) {
151       if (i==8 || i==13 ||  i==18 || i==23) {
152         uuid[i] = '-';
153       } else {
154         if (rnd <= 0x02) rnd = 0x2000000 + (Math.random()*0x1000000)|0;
155         r = rnd & 0xf;
156         rnd = rnd >> 4;
157         uuid[i] = chars[(i == 19) ? (r & 0x3) | 0x8 : r];
158       }
159     }
160     return uuid.join('');
161   };
162 
163 })();
164 
165 exports.string = string;
