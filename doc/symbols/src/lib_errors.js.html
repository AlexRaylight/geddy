 18 var response = require('./response');
 19 
 20 var errors = new function () {
 21 
 22   var _this = this;
 23   this.errorTypes = {
 24     400: 'Bad Request',
 25     401: 'Unauthorized',
 26     404: 'Not Found',
 27     406: 'Not Acceptable',
 28     500: 'Internal Server Error'
 29   };
 30 
 31   var errorType;
 32   var errorConstructor;
 33   var createConstructor = function (code, errorType) {
 34     errorConstructor = function (message) {
 35       this.statusCode = code;
 36       this.statusText = errorType;
 37       this.message = message || errorType;
 38       Error.captureStackTrace(this);
 39     }
 40     errorConstructor.prototype = new Error();
 41     return errorConstructor;
 42   };
 43   for (var code in this.errorTypes) {
 44     // Strip spaces
 45     errorType = this.errorTypes[code].replace(/ /g, '');
 46     this[errorType + 'Error'] = createConstructor(code, errorType);
 47   }
 48   // For repetitively redundant name
 49   this.InternalServerError = this.InternalServerErrorError;
 50 
 51   this.respond = function (resp, e) {
 52     // The no-error error ... whoa, meta!
 53     if (!e) {
 54       throw new Error('No error to respond with.');
 55     }
 56     var r = new response.Response(resp);
 57     var code = e.statusCode || 500;
 58     var msg = '';
 59     if (geddy.config.detailedErrors) {
 60       msg = e.stack || e.message || String(e);
 61       msg = msg.replace(/\n/g, '<br/>');
 62     }
 63     msg = '<h3>Error: ' + code + ' ' +  _this.errorTypes[code] + '</h3>' + msg;
 64     r.send(msg, code, {'Content-Type': 'text/html'});
 65   };
 66 
 67 }();
 68 
 69 for (var p in errors) { exports[p] = errors[p]; }
