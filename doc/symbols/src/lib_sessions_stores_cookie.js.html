 19 var Cookie = function (callback) {
 20   this.setup(callback);
 21 };
 22 
 23 Cookie.prototype = new (function () {
 24   this.setup = function (callback) {
 25     callback();
 26   };
 27 
 28   this.read = function (session, callback) {
 29     var controller = session.controller
 30       , data = controller.cookies.get(geddy.config.cookieSessionKey);
 31     data = data ? JSON.parse(data) : {};
 32     callback(data);
 33   };
 34 
 35   this.write = function (session, callback) {
 36     var controller = session.controller
 37       , data = JSON.stringify(session.data);
 38     controller.cookies.set(geddy.config.cookieSessionKey, data);
 39     callback();
 40   };
 41 
 42 })();
 43 
 44 exports.Cookie = Cookie;
