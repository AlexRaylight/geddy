 24 var net = require('net')
 25   , http = require('http')
 26   , url = require('url')
 27   , fs = require('fs')

 32 var _modes = {
 33   UNSTARTED: 'unstarted'
 34 , STARTED: 'started'
 35 , RETIRED: 'retired'
 36 , SHUTDOWN: 'shutdown'
 37 };
 38 
 39 var Log = function (worker) {
 40   var self = this
 41     , types = [
 42         'debug'
 43       , 'info'
 44       , 'notice'
 45       , 'warning'
 46       , 'error'
 47       , 'critical'
 48       , 'alert'
 49       , 'emergency'
 50       , 'access'
 51       ]
 52     , type
 53     , loggerCreator = function (t) {
 54         return function (msg) {
 55           self.worker.sendMessage({method: 'log', logType: t, message: msg});
 56         };
 57       };
 58 
 59   this.worker = worker;
 60 
 61   for (var i = 0, ii = types.length; i < ii; i++) {
 62     type = types[i];
 63     this[type] = loggerCreator(type);
 64   }
 65 };

 67 /**
 68  * @constructor Creates instances of proxy-server workers that
 69  * handle HTTP requests
 70  */
 71 var ROUTE_TIMEOUT = 3000;
 72 var Worker = function () {
 73   this._geddyStarter = null;
 74   this.config = {};
 75   this.configured = false;
 76   this.allowedOrigins = {};
 77   this.server = http.createServer();
 78   this.mode = _modes.UNSTARTED;
 79   this.requestTime = null;
 80   this.inFlight = {};
 81   this.handlerId = 0;
 82   this.inFlightIntervalId = null;
 83   this.messageParser = null;
 84   this.handlerTimers = {};
 85   this.log = new Log(this);
 86   this.stdin = null;
 87   this.init();
 88   // A shared object that all handlers running on the worker can use, if they want
 89   this.shared = {};
 90   this.readyQueue = [];
 91 };
 92 
 93 Worker.prototype = new function () {
 94 
 95   this.onReady = function onReady(fn) {
 96     if (this.configured) {
 97       fn();
 98     } else {
 99       this.readyQueue.push(fn);
100     }
101   };
102 
103   this.shutdownServer = function shutdownServer() {
104     this.sendMessage({method: 'shutdown'});
105   };
106 
107   /* Wait for our routes to be set */
108   this.init = function () {
109     var self = this;
110       /*
111       , areWeReadyYet = setInterval(function () {
112         if (self.router && self.router.routes) {
113           clearInterval(areWeReadyYet);
114           clearTimeout(noRoutes);
115           self.start();
116         }
117       }, 200)
118       , noRoutes = setTimeout(function () {
119         self.log.error('No routes defined after 30 seconds.  Exiting...');
120         process.exit(1);
121       }, ROUTE_TIMEOUT);
122       */
123 
124     // Global handle for error handling.
125     process.worker = self;
126 
127     // This process listens on stdin to communicate with the parent.
128     self.stdin = new net.Stream(0, 'unix');
129 
130     // Setup our server process,
131     // Listen on the passed-in shared FD and open it
132     self.stdin.addListener('fd', function (fd) {
133       if (self.mode == _modes.UNSTARTED) {
134         self.mode = _modes.STARTED;
135         self.server.listenFD(fd, 'tcp4');
136       }
137     });
138     self.stdin.addListener('data', function (data) {
139       self.receiveMessage(data);
140     });
141     // We resume stdin on start()
142   };
143 
144   this.configure = function (cfg) {
145     this.config = cfg;
146     this.configured = true;
147 148     for(var i = 0, ii = this.readyQueue.length; i < ii; ++i) {
149       this.readyQueue[i]();
150     }
151     this._geddyStarter(cfg);
152   };
153 
154   /*  Functions that can be called from Master Server  */
155   var _dispatch = {
156     config: function (msg) {
157       this.configure(msg.data);
158       this.log.notice('Server worker running in ' + this.config.environment + ' with a PID of: ' + process.pid);
159       this.log.debug('LOGGING STARTED ============================================');
160       this.log.debug('============================================================');
161     }
162   , retire: function (msg) {
163       var self = this;
164       this.server.addListener('close', function () {
165         self.mode = _modes.RETIRED;
166         self.watchInFlight();
167       });
168       this.server.close();
169     }
170   , shutdown: function (msg) {
171       this.mode = _modes.SHUTDOWN;
172       this.watchInFlight();
173     }
174   };
175 176 

177   /*  Process Communication  */
178 
179   this.receiveMessage = function (d) {
180     this.messageParser.handle(d);
181   };
182 
183   this.sendMessage = function (msg) {
184       var output = JSON.stringify(msg);
185       process.stdout.write(output + '\n');
186   };
187 
188 
189 
190   /*  Gracefully kill requests that take too long during shutdown */
191   this.watchInFlight = function () {
192     var self = this
193       , start = (new Date()).getTime();
194     this.inFlightIntervalId = setInterval(function () {
195       var reg = self.inFlight
196         , found
197         , timedOut = (new Date()).getTime() - start > self.config.gracefulRestartTimeout;
198       if (timedOut) {
199         for (var p in reg) {
200           reg[p].serverResponse.end();
201         }
202         found = false;
203       }
204       else {
205         for (var p in reg) {
206           found = true;
207 208         }
209       }
210       if (!found) {
211         clearInterval(self.inFlightIntervalId);
212         // Send the shutdown message
213         self.sendMessage({method: self.mode});
214       }
215     }, 2000);
216   };
217 
218 
219   /*  Begin handling requests  */
220   this.start = function (callback) {
221     this._geddyStarter = callback;
222     var self = this;
223     this.messageParser = new utils.MessageParser(_dispatch, this);
224 
225 226     // Tell the master we're alive
227     setInterval(function () {
228       self.sendMessage({method: 'heartbeat'});
229     }, 5000);
230 
231     // Get the hostname
232     this.hostname = '';
233     hostproc = spawn('hostname');
234     hostproc.stdout.on('data', function (data) {
235       self.hostname += data;
236     });
237     hostproc.stdout.on('end', function () {
238       self.hostname = self.hostname.replace('/\s+$/', '');
239     });
240 
241 
242     self.stdin.resume();
243   };
244 };
245 
246 exports.Worker = Worker;
