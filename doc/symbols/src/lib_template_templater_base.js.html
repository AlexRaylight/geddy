 19 var util = require('util'),
 20     EventEmitter = require('events').EventEmitter;
 21 
 22 /**
 23  * A TemplaterBase is an instance of EventEmitter that emits a 'data'
 24  * and an 'end' event
 25  * @contstructor
 26  */
 27 var TemplaterBase = function () {};
 28 
 29 TemplaterBase.prototype = new EventEmitter();
 30 
 31 TemplaterBase.prototype.eventTypes = {
 32   DATA: 'data'
 33   , END: 'end'
 34 };
 35 
 36 /**
 37  * The render method renders the data into a template string,
 38  * emiting a 'data' event whenever there's data to write to the
 39  * response
 40  *
 41  * @param {Object} data The object to use for the source of data
 42  * in the rendered template
 43  * @param {Array} [paths] An optional list of paths to use to look
 44  * for template-partials. By default, this will contain a single item,
 45  * the templateRoot of the calling controller.
 46  * @param {String} [paths] An optional filename to get the initial 
 47  * template partial. By default, this will contain the name of the
 48  * action called on the controller.
 49  */
 50 TemplaterBase.prototype.render = function (data, paths, filename) {
 51   
 52   // Do some stuff with the passed-in data
 53   var content = util.inspect(data);
 54 
 55   // Emit 'data' events whenever there's content to push out to
 56   // the response
 57   this.emit(this.eventTypes.DATA, content);
 58   
 59   // Emit 'end' when all finished
 60   this.emit(this.eventTypes.END);
 61 };
 62 
 63 exports.TemplaterBase = TemplaterBase;
