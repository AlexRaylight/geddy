 26 ejs.TemplateNode = function (id, url, params, parentNode) {
 27   this.id = id;
 28   this.url = url;
 29   this.dirname = path.dirname(url);
 30   this.params = params;
 31   this.parentNode = parentNode || null;
 32   this.loaded = false;
 33   this.cachedFinish = false;
 34   this.childNodes = {};
 35   this.content = null;
 36 };
 37 
 38 ejs.TemplateNode.prototype = new function () {
 39   // Only the root template node gets this -- it's the function that
 40   // passes the completed content back to be written out in the response
 41   this.finishRoot = null;
 42 
 43   this.childFinished = function (childNode) {
 44     this.content = this.content.replace(
 45         '###partial###' + childNode.id, childNode.content);
 46 
 47     // Every time a child node finished, this node may actually
 48     // be finished
 49     this.tryFinish();
 50   };
 51 
 52   this.tryFinish = function () {
 53     // If this node is actually finished, either call up the
 54     // chain, or finish the entire thing out
 55     if (this.isFinished()) {
 56       if (this.parentNode) {
 57         this.parentNode.childFinished(this);
 58       }
 59       else {
 60         this.finishRoot();
 61       }
 62     }
 63   };
 64 
 65   var _createLoader = function (node) {
 66     return function () { node.loadTemplate.call(node); };
 67   };
 68 
 69   this.loadChildTemplates = function () {
 70     var childNodes = this.childNodes;
 71     var childNode;
 72     for (var p in childNodes) {
 73       childNode = childNodes[p];
 74       utils.async.execNonBlocking(_createLoader(childNode));
 75     }
 76   };
 77 
 78   this.isFinished = function () {
 79     // May be a recursive check down into a child -- if the child
 80     // is not loaded, it's not finished
 81     if (!this.loaded) {
 82       return false;
 83     }
 84 
 85     // If we've already closed this node out, use the cached finish-flag
 86     if (this.cachedFinish) {
 87       return true;
 88     }
 89 
 90     // If any of the child nodes of this parent haven't loaded yet,
 91     // bail out -- this will get re-called every time one of them
 92     // comes in, until they're all there and we can move on up the chain
 93     var childNodes = this.childNodes;
 94     for (var p in childNodes) {
 95       if (!childNodes[p].isFinished()) {
 96         return false;
 97       }
 98     }
 99     // Once this guy is finished cache the results to avoid
100 
101     // a bunch of recursive calls when it's called as a child
102     this.cachedFinish = true;
103 
104     return true;
105   };
106 
107   this.loadTemplate = function () {
108     var _this = this;
109 
110     var handleLoaded = function (data) {
111       _this.loaded = true;
112 
113       // Create a template out of the markup in the file
114       var templ = new ejs.Template({text: data});
115 
116       // Execute the template with the content as the data-object
117       // This may execute "partial" method calls that create child
118       // nodes for this one.
119       templ.process({data: _this.params});
120 
121       _this.content = templ.markup;
122 
123       // By the time we get here, all 'partial' method calls will
124       // have executed, and any children will have been added
125 126       _this.loadChildTemplates();
127 
128       // This node is loaded, and we now know if it has any children
129       // -- it may be finished, too.
130       _this.tryFinish();
131     };
132 
133     var cached = ejs.textCache[this.url];
134     // Use cached template text if possible
135     if (geddy.config.environment == 'production' && cached) {
136       handleLoaded(cached);
137     }
138     // If this is the first hit, use the template text off disc
139     // and cache it for subsequent requests
140     else {
141       fs.readFile(this.url, 'utf8', function (err, data) {
142         if (err) { throw err; }
143         ejs.textCache[_this.url] = data;
144         handleLoaded(data);
145       });
146     }
147 
148   };
149 
150 }();

152 ejs.Template = function (p) {
153   var UNDEF;
154   var params = p || {};
155 
156   this.mode = null;
157   this.templateText = params.text ||
158     // If you don't want to use Fleegix.js,
159     // override getTemplateTextFromNode to use
160     // textarea node value for template text
161     this.getTemplateTextFromNode(params.node);
162   this.afterLoaded = params.afterLoaded;
163   this.source = '';
164   this.markup = UNDEF;
165   // Try to get from URL
166   if (typeof this.templateText == 'undefined') {
167     // If you don't want to use Fleegix.js,
168     // override getTemplateTextFromUrl to use
169     // files for template text
170     this.getTemplateTextFromUrl(params);
171   }
172 };
173 
174 ejs.Template.prototype = new function () {
175   var _REGEX = /(<%%)|(%%>)|(<%=)|(<%#)|(<%)|(%>\n)|(%>)|(\n)/;
176   this.modes = {
177     EVAL: 'eval',
178     OUTPUT: 'output',
179     APPEND: 'append',
180     COMMENT: 'comment',
181     LITERAL: 'literal'
182   };
183   this.getTemplateTextFromNode = function (node) {
184     // Requires the fleegix.xhr module
185     if (typeof fleegix.string == 'undefined') {
186       throw('Requires fleegix.string module.'); }
187     var ret;
188     if (node) {
189       ret = node.value;
190       ret = fleegix.string.unescapeXML(ret);
191       ret = fleegix.string.trim(ret);
192     }
193     return ret;
194   };
195   this.getTemplateTextFromUrl = function (params) {
196     // Requires the fleegix.xhr module
197     if (typeof fleegix.xhr == 'undefined') {
198       throw('Requires fleegix.xhr module.'); }
199     var _this = this;
200     var url = params.url;
201     var noCache = params.preventCache || false;
202     // Found text in cache, and caching is turned on
203     if (text && !noCache) {
204       this.templateText = text;
205     }
206     // Otherwise go grab the text
207     else {
208       // Callback for setting templateText and caching --
209       // used for both sync and async loading
210       var callback = function (s) {
211 212         _this.templateText = s;
213         ejs.templateTextCache[url] = s;
214         // Use afterLoaded hook if set
215         if (typeof _this.afterLoaded == 'function') {
216           _this.afterLoaded();
217         }
218       };
219       var opts;
220       if (params.async) {
221         opts = {
222           url: url,
223           method: 'GET',
224           preventCache: noCache,
225           async: true,
226           handleSuccess: callback
227         };
228         // Get templ text asynchronously, wait for
229         // loading to exec the callback
230         fleegix.xhr.send(opts);
231       }
232       else {
233         opts = {
234           url: url,
235           method: 'GET',
236           preventCache: noCache,
237           async: false
238         };
239         // Get the templ text inline and pass directly to
240         // the callback
241         text = fleegix.xhr.send(opts);
242         callback(text);
243       }
244     }
245   };
246   this.process = function (p) {
247     var params = p || {};
248     this.data = params.data || {};
249     var domNode = params.node;
250     // Cache/reuse the generated template source for speed
251     this.source = this.source || '';
252     if (!this.source) { this.generateSource(); }
253 
254     // Eval the template with the passed data
255     // Use 'with' to give local scoping to data obj props
256     // ========================
257     var _output = ''; // Inner scope var for eval output
258     with (this.data) {
259       eval(this.source);
260     }
261     this.markup = _output;
262 
263     if (domNode) {
264       domNode.innerHTML = this.markup;
265     }
266     return this.markup;
267   };
268   this.generateSource = function () {
269     var line = '';
270     var matches = this.parseTemplateText();
271     if (matches) {
272       for (var i = 0; i < matches.length; i++) {
273         line = matches[i];
274         if (line) {
275           this.scanLine(line);
276         }
277       }
278     }
279   };
280   this.parseTemplateText = function() {
281     var str = this.templateText;
282     var pat = _REGEX;
283     var result = pat.exec(str);
284     var arr = [];
285     while (result) {
286       var firstPos = result.index;
287       var lastPos = pat.lastIndex;
288       if (firstPos !== 0) {
289         arr.push(str.substring(0, firstPos));
290         str = str.slice(firstPos);
291       }
292       arr.push(result[0]);
293       str = str.slice(result[0].length);
294       result = pat.exec(str);
295     }
296     if (str !== '') {
297       arr.push(str);
298     }
299     return arr;
300   };
301   this.scanLine = function (line) {
302     var _this = this;
303     var _addOutput = function () {
304       line = line.replace(/\n/, '\\n');
305       line = line.replace(/"/g, '\\"');
306       _this.source += '_output += "' + line + '";';
307     };
308     switch (line) {
309       case '<%':
310         this.mode = this.modes.EVAL;
311         break;
312       case '<%=':
313         this.mode = this.modes.OUTPUT;
314         break;
315       case '<%#':
316         this.mode = this.modes.COMMENT;
317         break;
318       case '<%%':
319         this.mode = this.modes.LITERAL;
320         this.source += '_output += "' + line + '";';
321         break;
322       case '%>':
323       case '%>\n':
324         if (this.mode == this.modes.LITERAL) {
325           _addOutput();
326         }
327         this.mode = null;
328         break;
329       default:
330         // In script mode, depends on type of tag
331         if (this.mode) {
332           switch (this.mode) {
333             // Just executing code
334             case this.modes.EVAL:
335               this.source += line;
336               break;
337             // Exec and output
338             case this.modes.OUTPUT:
339               // Add the exec'd result to the output
340               this.source += '_output += ' + line + ';';
341               break;
342             case this.modes.COMMENT:
343               // Do nothing
344               break;
345             // Literal <%% mode, append as raw output
346             case this.modes.LITERAL:
347               _addOutput();
348               break;
349           }
350         }
351         // In string mode, just add the output
352         else {
353           _addOutput();
354         }
355     }
356   };
357 };
