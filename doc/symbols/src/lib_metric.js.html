 18 /**
 19 * This is an adapter to send metrics messages from messagePassers to the master process
 20 * messagePasser is likely a messagePasser
 21 */
 22 
 23 // TODO: if metrics are turned off, make all of these noops
 24 
 25 var Metric = exports = module.exports = function Metrics(messagePasser, eventType) {
 26   this.messagePasser = messagePasser;
 27   this.eventType = eventType;
 28 }
 29 
 30 Metric.prototype.newMetric = function(type, eventType) {
 31   this.messagePasser.sendMessage({
 32       method: 'createMetric'
 33     , type: type
 34     , eventType: eventType
 35   });
 36 }
 37 
 38 Metric.prototype.forwardMessage = function(method, args) { 
 39   this.messagePasser.sendMessage({
 40       method: 'updateMetric'
 41     , metricMethod: method
 42     , metricArgs: args
 43     , eventType: this.eventType
 44   });
 45 }
 46 
 47 Metric.prototype.update = function(val) { return this.forwardMessage('update', [val]); }
 48 Metric.prototype.mark = function(n) { return this.forwardMessage('mark', [n]); }
 49 Metric.prototype.inc = function(n) { return this.forwardMessage('inc', [n]); }
 50 Metric.prototype.dec = function(n) { return this.forwardMessage('dec', [n]); }
 51 Metric.prototype.clear = function() { return this.forwardMessage('clear'); }
 52 
 53 // type must be one of: Timer, Histogram, Meter, Counter
 54 module.exports.new = function(messagePasser, eventType, type) {
 55   var metric = new Metric(messagePasser, eventType);
 56   metric.newMetric(type, eventType);
 57   return metric;
 58 }
