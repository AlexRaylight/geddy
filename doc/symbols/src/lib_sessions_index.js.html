 19 var session = new (function () {
 20   var KEY_LENGTH = 32;
 21 
 22   this.store = null;
 23 
 24   this.generateSessionId = function () {
 25     var chars = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXTZabcdefghiklmnopqrstuvwxyz';
 26     var len = KEY_LENGTH;
 27     var str = '';
 28     var mls = new Date().getTime();
 29     for (var i = 0; i < len; i++) {
 30       var rnum = (Math.random() * chars.length);
 31       rnum = Math.floor(rnum);
 32       str += chars.substring(rnum, rnum + 1);
 33     }
 34     return str;
 35   };
 36 
 37   this.createStore = function (type, callback) {
 38     var key = geddy.string.capitalize(type);
 39     var constructor = require('./stores/' + type)[key];
 40     session.store = new constructor(callback);
 41   };
 42 
 43 })();
 44 
 45 session.Session = function (controller, callback) {
 46   var self = this
 47     , cookies = controller.cookies
 48     , keyName = geddy.config.sessions.key
 49     , sid = cookies.get(keyName);
 50 
 51   if (!sid) {
 52     sid = session.generateSessionId()
 53     var dt = new Date();
 54     dt.setTime(dt.getTime() + (geddy.config.sessions.expiry * 1000));
 55     cookies.set(keyName, sid, {expires: dt.toGMTString()});
 56   }
 57 
 58   this.sid = sid;
 59   this.controller = controller;
 60   this.data = null;
 61 
 62   setTimeout(function () {
 63     self.init(callback);
 64   }, 0);
 65 };
 66 
 67 session.Session.prototype = new function () {
 68   this.init = function(appCallback) {
 69     var _this = this;
 70     var localCallback = function (result) {
 71       _this.data = result;
 72       appCallback();
 73     };
 74     session.store.read(this, localCallback);
 75   };
 76 
 77   this.get = function (key) {
 78     return this.data[key];
 79   };
 80 
 81   this.set = function (key, val) {
 82     this.data[key] = val;
 83   };
 84 
 85   this.unset = function (key) {
 86     delete this.data[key];
 87   };
 88 
 89   this.close = function (appCallback) {
 90     session.store.write(this, appCallback);
 91   };
 92 
 93 }();
 94 
 95 module.exports = session;
